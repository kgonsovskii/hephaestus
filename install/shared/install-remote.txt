export DEBIAN_FRONTEND=noninteractive
: "${HEPHAESTUS_PROFILE:?HEPHAESTUS_PROFILE must be set for remote install}"
CLONE_DIR="${HOME}/hephaestus"
PROFILE_FILE="$(dirname "${CLONE_DIR}")/profile.txt"
printf '%s\n' "${HEPHAESTUS_PROFILE}" > "${PROFILE_FILE}"
echo "[install-remote] profile ${HEPHAESTUS_PROFILE} -> ${PROFILE_FILE}"
wait_for_apt_dpkg_lock
apt_get update
apt_get install -y git ca-certificates
rm -rf "$CLONE_DIR"
git clone --depth 1 https://github.com/kgonsovskii/hephaestus.git "$CLONE_DIR"
bash "$CLONE_DIR/install/install.sh" "${HEPHAESTUS_PROFILE}"
